    <script>
        let categoryConfig = [
            { id: 'IMPORTANT', label: "🏦 IMPORTANT BILLS", icon: '🏦' },
            { id: 'STREAMING', label: "📺 STREAMING SUBS", icon: '📺' },
            { id: 'SHOPPING', label: "🛒 SHOPPING CARTS", icon: '🛒' },
            { id: 'PAYMENTS', label: "💳 PAYMENTS/OTHER", icon: '💳' }
        ];

        function renderCategoryManager() {
            const container = document.getElementById('categoryManager');
            container.innerHTML = categoryConfig.map((cat, index) => `
                <div class="flex items-center justify-between bg-slate-50 p-3 rounded-xl border border-slate-100">
                    <span class="font-bold text-sm">${cat.icon} ${cat.id}</span>
                    <button onclick="removeCategory(${index})" class="text-slate-400 hover:text-red-500 text-xs">Remove</button>
                </div>
            `).join('');
            
            const dropdowns = document.querySelectorAll('.cat-in');
            dropdowns.forEach(select => {
                const currentVal = select.value;
                select.innerHTML = categoryConfig.map(k => `<option value="${k.id}" ${k.id === currentVal ? 'selected' : ''}>${k.icon} ${k.id}</option>`).join('');
            });
        }

        function addCategory() {
            const nameIn = document.getElementById('newCatName');
            const iconIn = document.getElementById('newCatIcon');
            const name = nameIn.value.trim().toUpperCase();
            const icon = iconIn.value;

            if (!name) return;
            if (categoryConfig.find(c => c.id === name)) return;

            categoryConfig.push({ id: name, label: `${icon} ${name}`, icon: icon });
            nameIn.value = '';
            renderCategoryManager();
        }

        function removeCategory(index) {
            if (categoryConfig.length <= 1) return;
            categoryConfig.splice(index, 1);
            renderCategoryManager();
        }

        function addRow(name = "", amt = "", date = "", cat = "IMPORTANT") {
            const container = document.getElementById('billList');
            const row = document.createElement('div');
            row.className = 'flex flex-wrap gap-4 items-end bg-slate-50 p-6 rounded-2xl border border-slate-100 bill-item group animate-slide';
            
            if (!categoryConfig.find(c => c.id === cat)) cat = categoryConfig[0].id;

            row.innerHTML = `
                <div class="flex-[2] min-w-[200px]">
                    <label class="block text-[10px] font-bold text-slate-400 uppercase mb-1">Bill Name</label>
                    <input type="text" value="${name}" placeholder="e.g. Rent" class="name-in w-full p-3 rounded-xl border border-slate-200 bg-white focus:ring-2 focus:ring-blue-500 focus:outline-none">
                </div>
                <div class="w-32">
                    <label class="block text-[10px] font-bold text-slate-400 uppercase mb-1">Amount ($)</label>
                    <input type="number" value="${amt}" placeholder="0.00" class="amt-in w-full p-3 rounded-xl border border-slate-200 bg-white focus:ring-2 focus:ring-blue-500 focus:outline-none">
                </div>
                <div class="w-32">
                    <label class="block text-[10px] font-bold text-slate-400 uppercase mb-1">Due Date</label>
                    <input type="text" value="${date}" placeholder="e.g. 1st" class="date-in w-full p-3 rounded-xl border border-slate-200 bg-white focus:ring-2 focus:ring-blue-500 focus:outline-none">
                </div>
                <div class="flex-1 min-w-[150px]">
                    <label class="block text-[10px] font-bold text-slate-400 uppercase mb-1">Category</label>
                    <select class="cat-in w-full p-3 rounded-xl border border-slate-200 bg-white appearance-none cursor-pointer">
                        ${categoryConfig.map(k => `<option value="${k.id}" ${k.id === cat ? 'selected' : ''}>${k.icon} ${k.id}</option>`).join('')}
                    </select>
                </div>
                <button onclick="this.parentElement.remove()" class="p-3 text-slate-300 hover:text-red-500 transition-colors">✕</button>
            `;
            container.appendChild(row);
        }

        window.onload = () => {
            renderCategoryManager();
            addRow("Rent", "152.00", "1st", "IMPORTANT");
            addRow("Netflix", "24.99", "1st", "STREAMING");
            addRow("Harbor Freight", "24.98", "25th", "SHOPPING");

        let finalBlueprint = {};
        let pythonSnippets = "";

        function generateBlueprint() {
            const items = document.querySelectorAll('.bill-item');
            const tbody = document.getElementById('blueprintRows');
            const resultArea = document.getElementById('resultArea');
            
            finalBlueprint = {};
            pythonSnippets = "";
            let currentSheetRow = 7;
            let subtotalCells = [];

            categoryConfig.forEach(cat => {
                finalBlueprint[currentSheetRow] = `${cat.label}\t\t\t\t\t`;
                currentSheetRow++;

                let startBillRow = currentSheetRow;
                let count = 0;
                items.forEach(item => {
                    if (item.querySelector('.cat-in').value === cat.id) {
                        const name = item.querySelector('.name-in').value || "Unnamed";
                        const amt = item.querySelector('.amt-in').value || "0.00";
                        const date = item.querySelector('.date-in').value || "";
                        const formula = `=IF(C${currentSheetRow},0,E${currentSheetRow})`;
                        finalBlueprint[currentSheetRow] = `${name}\t\tFALSE\t${formula}\t${amt}\t${date}`;
                        currentSheetRow++;
                        count++;
                    }
                });

                if (count === 0) {
                    finalBlueprint[currentSheetRow] = `(No ${cat.id} entries)\t\t\t\t\t`;
                    currentSheetRow++;
                }

                let endBillRow = currentSheetRow - 1;
                const subtotalFormula = `=SUM(D${startBillRow}:D${endBillRow})`;
                finalBlueprint[currentSheetRow] = `SUBTOTAL\t\t\t${subtotalFormula}\t\t`;
                subtotalCells.push(`D${currentSheetRow}`);
                
                pythonSnippets += `show_bill_list("${cat.id}", "${cat.icon}", "A${startBillRow}:C${endBillRow}")\n`;
                
                currentSheetRow += 2;
            });

            const sumSubtotals = subtotalCells.join(' + ');
            document.getElementById('g4Formula').innerText = `=D4 - (${sumSubtotals})`;
            document.getElementById('a4Formula').innerText = `=${sumSubtotals}`;
            document.getElementById('pythonCode').innerText = pythonSnippets;

            tbody.innerHTML = '';
            const allRows = Object.keys(finalBlueprint).sort((a,b) => a-b);
            allRows.forEach(rowIdx => {
                const parts = finalBlueprint[rowIdx].split('\t');
                const isHeader = categoryConfig.some(c => parts[0] === c.label);
                const isSub = parts[0] === 'SUBTOTAL';
                const rowClass = isHeader ? 'bg-slate-900 text-white font-bold' : (isSub ? 'bg-blue-100 font-black' : 'hover:bg-slate-50 border-b');
                
                tbody.innerHTML += `
                    <tr class="${rowClass}">
                        <td class="p-2 font-mono text-[10px] text-slate-400">${rowIdx}</td>
                        <td class="p-2">${parts[0]}</td>
                        <td class="p-2 font-bold ${parts[2] === 'FALSE' ? 'text-blue-500' : ''}">${parts[2] || ''}</td>
                        <td class="p-2 font-mono text-[10px] text-blue-700 font-bold">${parts[3] || ''}</td>
                        <td class="p-2">${parts[4] ? '$' + parseFloat(parts[4]).toFixed(2) : ''}</td>
                        <td class="p-2">${parts[5] || ''}</td>
                    </tr>
                `;
            });

            resultArea.classList.remove('hidden');
            resultArea.scrollIntoView({ behavior: 'smooth' });
        }

        function copyToClipboard(text, msg) {
            const buffer = document.getElementById('clipboardBuffer');
            buffer.value = text;
            buffer.select();
            document.execCommand('copy');
            showToast(msg + " Copied!");
        }

        function copyAllData() {
            let tsv = "";
            const maxRow = Math.max(...Object.keys(finalBlueprint));
            for (let i = 7; i <= maxRow; i++) {
                const data = finalBlueprint[i] || "\t\t\t\t\t";
                tsv += data + "\n";
            }
            const buffer = document.getElementById('clipboardBuffer');
            buffer.value = tsv;
            buffer.select();
            document.execCommand('copy');
            showToast("Blueprint Copied!");
        }

        function showToast(msg) {
            const toast = document.getElementById('toast');
            toast.innerText = msg;
            toast.classList.remove('hidden');
            setTimeout(() => toast.classList.add('hidden'), 3500);
        }
